PARA CREAR TABLA

  • touch flight.sql | crea el archivo sql
  • CREATE TABLE flight () | creando tabla de nombre vuelos
  • id INTEGER PRIMARY KEY AUTO_INCREMENT, | primera columna tipo ID, y automatico
  • origin TEXT NOT NULL, | primera columna, tipo texto y no puede estar vacia
  • destination TEXT NOT NULL, | segunda columna, tipo texto y no puede estar vacia
  • duration INTEGER NOT NULL); | tercera columna, tipo integer y no puede estar vacia
  • luego se le da click derecho y run selected sql y se crea la tabla

PARA AGREGAR DATA A LA TABLA o VERLOS

  • INSERT INTO flight (origin, destination, duration) | se selecciona la tabla y parametros a los que le vamos a agregar
  • VALUES ("New york", "London", 356); | se le da valores a las celdas en el orden de las columnas
  • nuevamente click derecho, run selected SQL
  • SELECT * FROM flight; | muestra la info de la tabla flight de todas (*) las columnas, ( SIEMPRE CLICK DERECHO Y RUN SELECTED SQL)
  • SELECT * FROM flight WHERE origin="Brasilia"; | muestra unicamente las filas de la tabla que tienen como origen Brasilia
  • UPDATE flight SET origin="Buenos Aires" WHERE id=3; | cambia un dato del id 3 y columna origin, por el que le damos
  • SELECT * FROM flight WHERE duration > 400; | podemos poner condiciones tambien
  • SELECT * FROM flight WHERE origin LIKE "%a%"; | el "%a%" se usa para buscar lo que hay entre % en esa columna
  • DELETE FROM flight WHERE origin != "Mexico"; | elimina las filas que cumplen con el criterio mencionado
  • LIMIT, ORDER BY, GROUP BY, HAVING | devuelve resultado filtrado
  • SELECT * FROM flight WHERE origin LIKE "%a%" ORDER BY duration;
  • SELECT nombre,origin_id,destination_id FROM vuelo A INNER JOIN pasajero B ON A.flight_id=B.flight_id; | aca tenemos 2 tablas que tienen flight_id, entonces mostramos un resultado que las une
CREATE TABLE flight ( id INTEGER PRIMARY KEY AUTOINCREMENT, origin TEXT NOT NULL, destination TEXT NOT NULL, duration INTEGER NOT NULL, );
